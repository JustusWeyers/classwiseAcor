# Plot definition
plot_acor = function(r_acor, cpp_acor) {
  plot(x = 1:nrow(r_acor), y = r_acor$Autocorrelation)
  points(x = 1:nrow(cpp_acor), y = cpp_acor$Autocorrelation, pch = 2)
}

# Make a plot
plot_acor(r_acor, cpp_acor)
